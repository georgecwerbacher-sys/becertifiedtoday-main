json_file = open("input.json", "r")
json_file_content = json_file.()

decoded_json["interface"]["ip"] = "10.10.10.10"
encoded_json_compact = json.(decoded_json)
json_file = open("output.json", "w")
json_file.(encoded_json_compact)
